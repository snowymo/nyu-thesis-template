\begin{definition}
  A function $f$ is said to be \emph{continuous} if its derivative exists at every point.
\end{definition}

\begin{lemma}
  Let $f$ be a function whose derivative exists in every point, then $f$ is 
  a continuous function.
\end{lemma}
